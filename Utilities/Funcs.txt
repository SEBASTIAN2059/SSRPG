// Tecnicas

func Moondial(w1,w2)
 ?totaltime%3=0
  equipL moon
  equipR @w1@
 :?totaltime%3=1
  equipL moon
  equipR @w2@
 :?totaltime%3=2
  equipR @w1@
  equipL @w2@


func LongRangeSpit(x)
 ?screen.i=(x-34)/69 &
 ^item.GetCooldown(＂mask＂)=0

  ?screen.FromWorldX(x)<69
   Mask()
   screen.ResetOffset()
  :
   screen.Next()


func Exec(k,v)
 ?item.CanActivate() & (item.GetCooldown(k)<=0
 ^| item.right = v & item.right.state = 2)
  equip @v@
  activate R
  return True
 return False

func Exec2(k, v, lvl)
 ?item.CanActivate() & (item.GetCooldown(k)<=0
 ^| item.right = v & item.right.state = 2)
  equip @v@ @lvl@
  activate R
  return True
 return False

// BLADE
var waitToBlade = 0

func Blade(next)
 ?item.GetCooldown(＂blade＂) <= 0 & item.CanActivate()
  ?next
   waitToBlade++
   screen.Next()
   ?waitToBlade>18
    equip blade
    activate R
    screen.ResetOffset()
    waitToBlade=0
  :
   equip blade
   activate R
  return true
 return false
