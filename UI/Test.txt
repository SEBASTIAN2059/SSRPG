var sub_panel_options = []
var sub_panel_option_texts = ["General","Player","Foe","Time","Foe B/D","Piedras"]

func open_panel(btn)
   play key_drop
   btn.visible = !btn.visible
   panel.visible = !panel.visible
   sub_panel.visible = false
   storage.Set("primary_ui", panel.visible)
   storage.Set("primary_button", btn.visible)

func close_panel()
   play key_drop
   primary_button.visible = !primary_button.visible
   panel.visible = !panel.visible
   sub_panel.visible = inherit
   storage.Set("primary_ui", panel.visible)
   storage.Set("primary_button", primary_button.visible)

func press_popup_button(btn)
   play key_drop
   //Implement popup functionality here
   var anchor
   var dock
   var x

   ?btn = popup_button_left
    anchor = center_center
    dock = center_left
    x = segment_w/2 + margin * 2
    update_selected_option(text_1_data_key)
   :?btn = popup_button_center
    anchor = center_center
    dock = center_center
    x = 0
    update_selected_option(text_2_data_key)
   :?btn = popup_button_right
    anchor = center_center
    dock = center_right
    x = -segment_w/2 - margin * 2
    update_selected_option(text_3_data_key)

   ?last_popup_button = btn
    sub_panel.visible = !sub_panel.visible
   :
    last_popup_button = btn
    sub_panel.visible = true
    sub_panel.anchor = anchor
    sub_panel.dock = dock
    sub_panel.x = x
    
func update_selected_option(index)
   for i = 0..options - 1
    ?i = index
     sub_panel_options[i].tcolor = selected_text_color
    :
     sub_panel_options[i].tcolor = text_color

func press_option_button(btn)
   play key_drop
   //Implement option button functionality here
   btn.hcolor = color.Random()
   btn.tcolor = color.Random()
   var index = sub_panel_options.IndexOf(btn)
   ?last_popup_button = popup_button_left
    text_1_data_key = index
    popup_button_left.text = sub_panel_option_texts[index]
    storage.Set("text_1_id", index)
   ?last_popup_button = popup_button_center
    text_2_data_key = index
    popup_button_center.text = sub_panel_option_texts[index]
    storage.Set("text_2_id", index)
   ?last_popup_button = popup_button_right
    text_3_data_key = index
    popup_button_right.text = sub_panel_option_texts[index]
    storage.Set("text_3_id", index)
   sub_panel.visible = false

func create_ui_components()
   panel = create_panel(primary_ui_data_key, panel_color, pw, ph, 0, -margin, bottom_center, 3, null)

   sub_panel = create_panel(false, panel_color, sub_panel_w, sub_panel_h, 0, ph-margin*2, center_center, 2, panel)
   //In panel buttons and texts

   //Buttons
   close_button = create_button(

      ^top_center, top_center, 
      ^close_text_btn,

      ^0, panel, close_panel)

   //Popup panel button
   //left button
   popup_button_left = create_button(
      ^null,
      ^7, 1,
      ^segment_w/2 - margin, 0, 
      ^bottom_left, bottom_left, 
      ^sub_panel_option_texts[text_1_data_key],

   //center button
   popup_button_center = create_button(
      ^null,
      ^7, 1, 
      ^0, 0, 
      ^bottom_center, bottom_center, 
      ^sub_panel_option_texts[text_2_data_key], 
      ^panel_color, panel_color, hover_color, 
      ^0, panel, press_popup_button)
   //right button
   popup_button_right = create_button(
      ^null,
      ^7, 1, 
      ^-segment_w/2 + margin * 2, 0, 
      ^bottom_right, bottom_right, 
      ^sub_panel_option_texts[text_3_data_key], 
      ^panel_color, panel_color, hover_color, 
      ^0, panel, press_popup_button)

   //set options for sub panel
   for i = 1..options
    var option_button = create_button(
      ^null, 
      ^secundary_btn_w, secundary_btn_h, 
      ^0, margin + (secundary_btn_h * (i - 1)), 
      ^top_center, top_center, 
      ^sub_panel_option_texts[i-1], 
      ^panel_color, panel_color, hover_color, 
      ^0, sub_panel, press_option_button)
    sub_panel_options.Add(option_button)

   //Principal Button (open panel)
   primary_button = create_button(
      ^primary_button_data_key, 
      ^primary_btn_w, primary_btn_h, 
      ^0, margin, 
      ^top_center, top_center, 
      ^open_text_btn, 
      ^panel_color, panel_color, hover_color, 
      ^4, null, open_panel)

func get_dificult_color()
 var c
 ?loc.stars > 15
  c = green
 :?loc.stars > 10
  c = yellow
 :?loc.stars > 5
  c = cyan
 :
  c = white
 return c

func get_foe_element()
 var element
 ?foe = "poison"
  element = "Poison"
 :?foe = "vigor"
  element = "Vigor"
 :?foe = "ice"
  element = "Ice"
 :?foe = "aether"
  element = "Aether"
 :?foe = "fire"
  element = "Fire"
 :
    element = " "

 return element

func get_color_by_limit(limit)
 var good = #12ff34
 var bad = #ff3412
 var t = math.Min(loc.gp/limit, 1.0)
 var c = color.Lerp(good, bad, t)
 return c

func get_loc_gp_string()
 ?loc.stars ! 5 & loc.stars ! 10 & loc.stars ! 15 & loc.stars ! 20
  return ""
 var gp_string = "" 
 var mult = 1.0
 ?loc = "rocky"
   mult = 3.0

 ?loc.stars > 10
  gp_string = " GP: [color="+ #12ff34 +"]"+loc.gp+"[/color]/inf"
  return gp_string
 :
  ?loc.stars = 5
   gp_string = " GP: [color="+ get_color_by_limit(150.0 * mult) +"]"+loc.gp+"[/color]/"+150*mult
  :?loc.stars = 10
   gp_string = " GP: [color="+ get_color_by_limit(200.0 * mult) +"]"+loc.gp+"[/color]/"+200*mult
  return gp_string

func get_format_string(str)
 var s = string.Split(str, ",")
 var s2 = "Non"
 var format_s = []
 ?s.Count() = 0
  return ""
 for i = 0..s.Count() - 1
   var temp = string.Split(s[i], ":")
   ?temp.Count() = 1
    break
   var temp1 = [temp[2], temp[0]]
   var temp2 = [string.Join("", temp1), temp[3]]

   format_s.Add(string.Join(":", temp2))
   // Cleanup
   var temp = []
   var temp1 = []
   var temp2 = []
   
 s2 = string.Join(" ", format_s)
 return s2

func get_bd_string(str,f)
 ?str = "B"
   ?f
    ?foe.buffs.string
     return "\nBuffs:"+ str_color(get_format_string(foe.buffs.string), #c70000)
   :
    ?buffs.string
     return "\nBuffs:"+ str_color(get_format_string(buffs.string), #00c72b)
   return ""
 :?str = "D"
   ?f
    ?foe.debuffs.string
     return "\nDebuffs:"+ str_color(get_format_string(foe.debuffs.string), #00c72b)
   :
    ?debuffs.string
     return "\nDebuffs:"+ str_color(get_format_string(debuffs.string), #c70000)
   return ""

func str_color(text, color_code)
   return "[color="+ color_code +"]"+text+"[/color]"
